### 1.3.1 FrameEventConfig.cs — 配置基类

```csharp
using System;

/// <summary>
/// 帧事件配置基类（存在 ActionAsset SO 中，只读）。
/// 所有可变状态都在 CreateRuntime() 产生的 FrameEventRuntime 实例上。
/// </summary>
[Serializable]
public abstract class FrameEventConfig
{
    /// <summary>事件开始帧（含），从 0 开始。</summary>
    public int startFrame;
    
    /// <summary>事件结束帧（含）。对于瞬时事件（如 Impulse），endFrame == startFrame。</summary>
    public int endFrame;

    /// <summary>创建独立的运行时实例。每次播放 Action 都会 new 一份，保证多 Actor 互不干扰。</summary>
    public abstract FrameEventRuntime CreateRuntime();
}
```

### 1.3.2 FrameEventRuntime.cs — 运行时基类

```csharp
/// <summary>
/// 帧事件运行时基类（纯 C# 对象，每次播放 new）。
/// 承载所有可变状态，SO 上的 Config 永远只读。
/// 
/// 生命周期对照：
///   Timeline Clip       →  FrameEventRuntime
///   OnClipStart          →  OnStart
///   OnClipUpdate         →  OnUpdate
///   OnClipStop(true)     →  OnStop(isFinish: true)
///   OnClipStop(false)    →  OnStop(isFinish: false)  或  ForceCleanup
/// </summary>
public abstract class FrameEventRuntime
{
    public enum Phase { Idle, Active, Finished }
    public Phase CurrentPhase { get; protected set; } = Phase.Idle;

    /// <summary>事件开始。对应 Timeline 的 OnClipStart。</summary>
    /// <param name="ctx">帧驱动上下文，包含 Actor、ActionInstance、当前帧信息等。</param>
    public abstract void OnStart(FrameDriverContext ctx);

    /// <summary>事件持续中的每帧更新。对应 Timeline 的 ProcessFrame / OnClipUpdate。</summary>
    public abstract void OnUpdate(FrameDriverContext ctx);

    /// <summary>事件正常结束或被动作切换打断。</summary>
    /// <param name="ctx">上下文</param>
    /// <param name="isFinish">true=帧号到达 endFrame 正常结束；false=动作被打断。</param>
    public abstract void OnStop(FrameDriverContext ctx, bool isFinish);

    /// <summary>强制清理（动作被打断时调用），保证不泄漏 GameObject 等资源。默认实现调用 OnStop(false)。</summary>
    public virtual void ForceCleanup(FrameDriverContext ctx)
    {
        if (CurrentPhase == Phase.Active)
            OnStop(ctx, false);
    }
}

/// <summary>
/// 帧驱动上下文，每帧由 FrameDriver 构建并传给所有 FrameEventRuntime。
/// 替代 Timeline 中 Behaviour 通过 PlayableGraph 间接获取的各种信息。
/// </summary>
public struct FrameDriverContext
{
    /// <summary>当前 Actor。</summary>
    public Actor Actor;
    
    /// <summary>当前 ActionInstance（可读取 EventContext、Config 等）。</summary>
    public ActionInstance ActionInstance;
    
    /// <summary>当前帧号（从 0 开始的整数）。</summary>
    public int CurrentFrame;
    
    /// <summary>当前事件的起始帧（从事件 Config 读取）。</summary>
    public int EventStartFrame;
    
    /// <summary>当前事件的结束帧。</summary>
    public int EventEndFrame;
    
    /// <summary>动作的帧率。</summary>
    public int FrameRate;
    
    /// <summary>当前播放速度（1.0 = 正常，0 = HitStop 冻结）。</summary>
    public float PlaybackSpeed;
    
    /// <summary>当前事件的归一化进度 (0..1)。Velocity 曲线采样用。</summary>
    public float NormalizedEventTime
    {
        get
        {
            int duration = EventEndFrame - EventStartFrame;
            if (duration <= 0) return 1f;
            return (float)(CurrentFrame - EventStartFrame) / duration;
        }
    }
    
    /// <summary>当前事件已经过的秒数（用于粒子模拟等需要浮点时间的场合）。</summary>
    public float EventElapsedSeconds => FrameRate > 0 
        ? (float)(CurrentFrame - EventStartFrame) / FrameRate 
        : 0f;
}
```

### 1.3.3 AnimEvent.cs — 替代 AnimancerClip + AnimancerBehaviour

```csharp
using System;
using UnityEngine;
using Animancer;

// ==================== Config ====================

[Serializable]
public class AnimEventConfig : FrameEventConfig
{
    [Tooltip("Drag a ClipTransition (attacks) or MixerTransition2D (8-way dodge).")]
    public TransitionAsset transitionAsset;
    public AnimancerParameterMode parameterMode;
    public Vector2 fallbackVector2;
    public float fallbackFloat;

    public override FrameEventRuntime CreateRuntime() => new AnimEventRuntime(this);
}

// ==================== Runtime ====================

public class AnimEventRuntime : FrameEventRuntime
{
    private readonly AnimEventConfig _cfg;
    private AnimancerState _currentState;

    public AnimEventRuntime(AnimEventConfig cfg) => _cfg = cfg;

    public override void OnStart(FrameDriverContext ctx)
    {
        if (_cfg.transitionAsset == null || _cfg.transitionAsset.Transition == null) return;
        var actor = ctx.Actor;
        if (actor == null || actor.animancer == null) return;

        _currentState = actor.animancer.Play(_cfg.transitionAsset.Transition);
        InitializeMixerParameter(ctx);

        if (_currentState != null)
        {
            _currentState.IsPlaying = true;
            // 同步起始时间：(当前帧 - 事件起始帧) / 帧率
            _currentState.Time = (float)(ctx.CurrentFrame - ctx.EventStartFrame) / ctx.FrameRate;
        }

        CurrentPhase = Phase.Active;
    }

    public override void OnUpdate(FrameDriverContext ctx)
    {
        if (_currentState == null) return;
        // 同步播放速度（HitStop 时 speed=0，动画自然冻结）
        _currentState.Speed = ctx.PlaybackSpeed;
    }

    public override void OnStop(FrameDriverContext ctx, bool isFinish)
    {
        if (_currentState != null)
            _currentState.IsPlaying = false;
        _currentState = null;
        CurrentPhase = Phase.Finished;
    }

    private void InitializeMixerParameter(FrameDriverContext ctx)
    {
        if (_currentState is MixerState<Vector2> mixer2D)
        {
            switch (_cfg.parameterMode)
            {
                case AnimancerParameterMode.ContextDirection2D:
                    mixer2D.Parameter = TryGetContextDirection2D(ctx, out var dir2D) ? dir2D : _cfg.fallbackVector2;
                    break;
                case AnimancerParameterMode.SerializedFallback:
                    mixer2D.Parameter = _cfg.fallbackVector2;
                    break;
            }
        }
        else if (_currentState is MixerState<float> mixer1D)
        {
            switch (_cfg.parameterMode)
            {
                case AnimancerParameterMode.ContextMagnitude:
                    mixer1D.Parameter = TryGetContextMagnitude(ctx, out var mag) ? mag : _cfg.fallbackFloat;
                    break;
                case AnimancerParameterMode.SerializedFallback:
                    mixer1D.Parameter = _cfg.fallbackFloat;
                    break;
            }
        }
    }

    private bool TryGetContextDirection2D(FrameDriverContext ctx, out Vector2 parameter)
    {
        parameter = _cfg.fallbackVector2;
        var inst = ctx.ActionInstance;
        var actor = ctx.Actor;
        if (inst == null || actor == null) return false;

        Vector3 direction = inst.EventContext.Direction;
        direction.y = 0f;
        if (direction.sqrMagnitude <= 0.0001f) return false;

        Vector3 localDirection = actor.transform.InverseTransformDirection(direction.normalized);
        parameter = new Vector2(localDirection.x, localDirection.z);
        if (parameter.sqrMagnitude <= 0.0001f) return false;

        parameter.Normalize();
        return true;
    }

    private bool TryGetContextMagnitude(FrameDriverContext ctx, out float magnitude)
    {
        magnitude = _cfg.fallbackFloat;
        var inst = ctx.ActionInstance;
        if (inst == null) return false;
        magnitude = inst.EventContext.Magnitude;
        return Mathf.Abs(magnitude) > 0.001f;
    }
}
```

### 1.3.4 ContinuousAnimEvent.cs — 替代 ContinuousAnimancerClip + Behaviour

```csharp
using System;
using UnityEngine;
using Animancer;

[Serializable]
public class ContinuousAnimEventConfig : FrameEventConfig
{
    public TransitionAsset transitionAsset;
    public ContinuousParameterSource parameterSource = ContinuousParameterSource.LocomotionIntent;
    [Min(0f)]
    public float verticalVelocitySmoothTime = 0.1f;

    public override FrameEventRuntime CreateRuntime() => new ContinuousAnimEventRuntime(this);
}

public class ContinuousAnimEventRuntime : FrameEventRuntime
{
    private readonly ContinuousAnimEventConfig _cfg;
    private AnimancerState _currentState;
    private float _smoothedVertVel;
    private float _smoothedVertVelVelocity;

    public ContinuousAnimEventRuntime(ContinuousAnimEventConfig cfg) => _cfg = cfg;

    public override void OnStart(FrameDriverContext ctx)
    {
        if (_cfg.transitionAsset == null || _cfg.transitionAsset.Transition == null) return;
        var actor = ctx.Actor;
        if (actor == null || actor.animancer == null) return;

        _currentState = actor.animancer.Play(_cfg.transitionAsset.Transition);

        if (_currentState is MixerState<Vector2> mixer2D)
            mixer2D.Parameter = Vector2.zero;
        else if (_currentState is MixerState<float> mixer1D)
            mixer1D.Parameter = 0f;

        if (actor.movement != null)
            _smoothedVertVel = actor.movement.CurrentVelocity.y;
        else
            _smoothedVertVel = 0f;
        _smoothedVertVelVelocity = 0f;

        if (_currentState != null)
        {
            _currentState.IsPlaying = true;
            _currentState.Time = (float)(ctx.CurrentFrame - ctx.EventStartFrame) / ctx.FrameRate;
        }

        CurrentPhase = Phase.Active;
    }

    public override void OnUpdate(FrameDriverContext ctx)
    {
        if (_currentState == null) return;
        _currentState.Speed = ctx.PlaybackSpeed;

        switch (_cfg.parameterSource)
        {
            case ContinuousParameterSource.LocomotionIntent:
                UpdateMixerFromLocomotionIntent(ctx.Actor);
                break;
            case ContinuousParameterSource.CharacterVelocity:
                UpdateMixerFromVelocity(ctx.Actor);
                break;
            case ContinuousParameterSource.VerticalVelocity:
                UpdateMixerFromVerticalVelocity(ctx.Actor);
                break;
        }
    }

    public override void OnStop(FrameDriverContext ctx, bool isFinish)
    {
        if (_currentState != null)
            _currentState.IsPlaying = false;
        _currentState = null;
        CurrentPhase = Phase.Finished;
    }

    // --- 以下三个方法与 ContinuousAnimancerBehaviour 完全相同，直接搬过来 ---

    private void UpdateMixerFromLocomotionIntent(Actor actor)
    {
        if (actor?.movement == null) return;
        var intent = actor.movement.LocomotionIntent;

        if (_currentState is MixerState<Vector2> mixer2D)
        {
            Vector3 referenceForward = intent.FacingDirection;
            referenceForward.y = 0f;
            if (referenceForward.sqrMagnitude < 0.0001f)
                referenceForward = actor.transform.forward;
            referenceForward.Normalize();

            Vector3 moveDir = intent.WorldMoveDirection;
            moveDir.y = 0f;
            if (moveDir.sqrMagnitude < 0.0001f || intent.MoveStrength < 0.01f)
            {
                mixer2D.Parameter = Vector2.zero;
                return;
            }
            moveDir.Normalize();
            Quaternion refRotation = Quaternion.LookRotation(referenceForward, Vector3.up);
            Vector3 localDir = Quaternion.Inverse(refRotation) * moveDir;
            Vector2 param = new Vector2(localDir.x, localDir.z) * intent.MoveStrength;
            if (param.sqrMagnitude > 1f) param.Normalize();
            mixer2D.Parameter = param;
        }
        else if (_currentState is MixerState<float> mixer1D)
        {
            mixer1D.Parameter = intent.MoveStrength;
        }
    }

    private void UpdateMixerFromVelocity(Actor actor)
    {
        if (actor?.movement == null) return;
        Vector3 velocity = actor.movement.CurrentVelocity;
        velocity.y = 0f;

        if (_currentState is MixerState<Vector2> mixer2D)
        {
            Vector3 localVel = actor.transform.InverseTransformDirection(velocity);
            float maxSpeed = actor.movement.LocomotionBaseSpeed;
            if (maxSpeed > 0.001f)
            {
                Vector2 param = new Vector2(localVel.x / maxSpeed, localVel.z / maxSpeed);
                if (param.sqrMagnitude > 1f) param.Normalize();
                mixer2D.Parameter = param;
            }
            else
            {
                mixer2D.Parameter = Vector2.zero;
            }
        }
        else if (_currentState is MixerState<float> mixer1D)
        {
            float speed = velocity.magnitude;
            float maxSpeed = actor.movement.LocomotionBaseSpeed;
            mixer1D.Parameter = maxSpeed > 0.001f ? speed / maxSpeed : 0f;
        }
    }

    private void UpdateMixerFromVerticalVelocity(Actor actor)
    {
        if (actor?.movement == null) return;
        if (_currentState is MixerState<float> mixer1D)
        {
            float rawVertVel = actor.movement.CurrentVelocity.y;
            if (_cfg.verticalVelocitySmoothTime > 0f)
            {
                _smoothedVertVel = Mathf.SmoothDamp(
                    _smoothedVertVel, rawVertVel,
                    ref _smoothedVertVelVelocity,
                    _cfg.verticalVelocitySmoothTime);
                mixer1D.Parameter = _smoothedVertVel;
            }
            else
            {
                mixer1D.Parameter = rawVertVel;
            }
        }
    }
}
```

### 1.3.5 HitBoxEvent.cs — 替代 ActionHitBoxClip + Behaviour

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

[Serializable]
public class HitBoxEventConfig : FrameEventConfig
{
    [Tooltip("骨骼路径（如 Armature/Hips/Spine/RightHand），运行时 actor.transform.Find 解析")]
    public string bonePath;
    public ActionHitBoxConfig hitboxConfig = new ActionHitBoxConfig();
    public AttackDataConfig dataConfig = new AttackDataConfig();

    [SerializeReference, SubclassSelector]
    public List<ImpactEffectConfig> effects = new List<ImpactEffectConfig>();

    public override FrameEventRuntime CreateRuntime() => new HitBoxEventRuntime(this);
}

public class HitBoxEventRuntime : FrameEventRuntime
{
    private readonly HitBoxEventConfig _cfg;
    private Transform _boneTransform;
    private GameObject _hitboxObject;
    private CapsuleCollider _collider;

    public HitBoxEventRuntime(HitBoxEventConfig cfg) => _cfg = cfg;

    public override void OnStart(FrameDriverContext ctx)
    {
        var actor = ctx.Actor;
        if (actor == null) return;

        // 解析骨骼路径（替代 ExposedReference<Transform>）
        if (!string.IsNullOrEmpty(_cfg.bonePath))
            _boneTransform = actor.transform.Find(_cfg.bonePath);

        if (_boneTransform == null) return;

        CreateHitbox(actor);
        CurrentPhase = Phase.Active;
    }

    public override void OnUpdate(FrameDriverContext ctx)
    {
        UpdateHitboxTransform();
    }

    public override void OnStop(FrameDriverContext ctx, bool isFinish)
    {
        DestroyHitbox();
        CurrentPhase = Phase.Finished;
    }

    private void CreateHitbox(Actor actor)
    {
        if (_hitboxObject != null) return;

        _hitboxObject = new GameObject("HitBox");
        _hitboxObject.hideFlags = HideFlags.HideInHierarchy;

        _collider = _hitboxObject.AddComponent<CapsuleCollider>();
        _collider.isTrigger = true;

        if (actor.combater != null)
        {
            var handler = _hitboxObject.AddComponent<AttackHandler>();
            handler.Init(actor.combater, _cfg.dataConfig);
            handler.RegisterHitStartEvent(this, OnHitStart);
            handler.RegisterHitOverEvent(this, OnHitOver);
        }

        UpdateHitboxTransform();
    }

    private void UpdateHitboxTransform()
    {
        if (_hitboxObject == null || _boneTransform == null) return;

        _hitboxObject.transform.position = _boneTransform.TransformPoint(_cfg.hitboxConfig.center);
        _hitboxObject.transform.rotation = _boneTransform.rotation * _cfg.hitboxConfig.rotation;

        if (_collider != null)
        {
            _collider.height = _cfg.hitboxConfig.height;
            _collider.radius = _cfg.hitboxConfig.radius;
            _collider.direction = 1;
        }
    }

    private void DestroyHitbox()
    {
        if (_hitboxObject == null) return;

        var handler = _hitboxObject.GetComponent<AttackHandler>();
        if (handler != null)
        {
            handler.UnregisterHitStartEvent(this);
            handler.UnregisterHitOverEvent(this);
        }

        if (Application.isPlaying)
            UnityEngine.Object.Destroy(_hitboxObject);
        else
            UnityEngine.Object.DestroyImmediate(_hitboxObject);

        _hitboxObject = null;
        _collider = null;
    }

    private void OnHitStart(AttackHitData data)
    {
        ImpactSystem.EnsureExists();
        if (ImpactSystem.Instance == null) return;

        var impactData = ImpactData.FromAttackHit(data);
        impactData.VfxSpawnPoint = data.HitPoint;
        impactData.FacingReferenceWorldPosition = HitVfxFacingUtility.ResolveFacingWorldPosition(
            impactData.TargetReceiver != null ? impactData.TargetReceiver.HitFacingTargetOverride : null,
            data.Attacker);
        Vector3 attackerRef = HitVfxAnchorUtility.GetDefaultAttackerRayOrigin(data.Attacker);
        impactData.PopulateDirectionalReferences(attackerRef);
        ImpactSystem.Instance.ApplyImpact(impactData, _cfg.effects);
    }

    private void OnHitOver(AttackHitData data) { }
}
```

### 1.3.6 VelocityEvent.cs — 替代 ActionVelocityClip + Behaviour

```csharp
using System;
using UnityEngine;

[Serializable]
public class VelocityEventConfig : FrameEventConfig
{
    public VelocityConfig config = new VelocityConfig();
    public override FrameEventRuntime CreateRuntime() => new VelocityEventRuntime(this);
}

public class VelocityEventRuntime : FrameEventRuntime
{
    private readonly VelocityEventConfig _cfg;
    private Transform _instigatorTransform;
    private float _savedGravityScale = 1f;

    public VelocityEventRuntime(VelocityEventConfig cfg) => _cfg = cfg;

    public override void OnStart(FrameDriverContext ctx)
    {
        var actor = ctx.Actor;
        if (actor?.movement == null || _cfg.config == null) return;

        _instigatorTransform = null;
        _savedGravityScale = 1f;
        actor.movement.SetGravityScale(_cfg.config.gravityScale);
        CurrentPhase = Phase.Active;
    }

    public override void OnUpdate(FrameDriverContext ctx)
    {
        var actor = ctx.Actor;
        if (actor?.movement == null || _cfg.config == null) return;

        var config = _cfg.config;
        // 归一化进度：纯整数运算，比 playable.GetTime()/GetDuration() 更精确
        float t = ctx.NormalizedEventTime;

        if (Mathf.Abs(config.horizontalSpeed) > 0.001f)
        {
            Vector3 dir = MotionDirectionResolver.Resolve(
                config.directionMode, actor, ctx.ActionInstance,
                config.fixedLocalDirection, ref _instigatorTransform);
            float hCurve = config.horizontalCurve?.Evaluate(t) ?? 1f;
            actor.movement.SetExternalHorizontalVelocity(dir * (config.horizontalSpeed * hCurve));
        }
        else
        {
            actor.movement.SetExternalHorizontalVelocity(Vector3.zero);
        }

        if (Mathf.Abs(config.verticalSpeed) > 0.001f)
        {
            float vCurve = config.verticalCurve?.Evaluate(t) ?? 1f;
            actor.movement.SetExternalVerticalVelocity(config.verticalSpeed * vCurve);
        }
        else
        {
            actor.movement.SetExternalVerticalVelocity(0f);
        }
    }

    public override void OnStop(FrameDriverContext ctx, bool isFinish)
    {
        var actor = ctx.Actor;
        if (actor?.movement != null)
        {
            actor.movement.ClearExternalVelocity();
            actor.movement.SetGravityScale(_savedGravityScale);
        }
        _instigatorTransform = null;
        CurrentPhase = Phase.Finished;
    }
}
```

### 1.3.7 ImpulseEvent.cs — 替代 ActionImpulseClip + Behaviour

```csharp
using System;
using UnityEngine;

[Serializable]
public class ImpulseEventConfig : FrameEventConfig
{
    public ImpulseConfig config = new ImpulseConfig();
    public override FrameEventRuntime CreateRuntime() => new ImpulseEventRuntime(this);
}

public class ImpulseEventRuntime : FrameEventRuntime
{
    private readonly ImpulseEventConfig _cfg;
    private Transform _instigatorTransform;
    private bool _didOverrideGravity;

    public ImpulseEventRuntime(ImpulseEventConfig cfg) => _cfg = cfg;

    public override void OnStart(FrameDriverContext ctx)
    {
        var actor = ctx.Actor;
        if (actor?.movement == null || _cfg.config == null) return;

        var config = _cfg.config;
        _instigatorTransform = null;
        Vector3 horizontalDir = MotionDirectionResolver.Resolve(
            config.directionMode, actor, ctx.ActionInstance,
            config.fixedLocalDirection, ref _instigatorTransform);

        if (Mathf.Abs(config.horizontalForce) > 0.001f)
            actor.movement.AddHorizontalImpulse(horizontalDir * config.horizontalForce);

        if (Mathf.Abs(config.verticalForce) > 0.001f)
            actor.movement.AddVerticalImpulse(config.verticalForce);

        _didOverrideGravity = false;
        if (config.gravityScale >= 0f)
        {
            actor.movement.SetGravityScale(config.gravityScale);
            _didOverrideGravity = true;
        }

        CurrentPhase = Phase.Active;
    }

    public override void OnUpdate(FrameDriverContext ctx)
    {
        // Impulse 一次性注入，Update 无逻辑（与原始 Behaviour 一致）
    }

    public override void OnStop(FrameDriverContext ctx, bool isFinish)
    {
        if (_didOverrideGravity && ctx.Actor?.movement != null)
        {
            ctx.Actor.movement.SetGravityScale(1f);
            _didOverrideGravity = false;
        }
        _instigatorTransform = null;
        CurrentPhase = Phase.Finished;
    }
}
```

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

[Serializable]
public class EffectEventConfig : FrameEventConfig
{

    [Tooltip("骨骼路径（如 Armature/Hips），空字符串=世界坐标")]

    public override FrameEventRuntime CreateRuntime() => new EffectEventRuntime(this);

public class EffectEventRuntime : FrameEventRuntime
{
    private readonly EffectEventConfig _cfg;
    private GameObject _particleInstance;
    private List<ParticleSystem> _particleSystems;
    private Transform _parentTransform;
    private bool _isPlaying;
    private float _lastTime = float.MaxValue;

    public EffectEventRuntime(EffectEventConfig cfg) => _cfg = cfg;

    public override void OnStart(FrameDriverContext ctx)
    {
        if (_cfg.particlePrefab == null) return;

        // 解析父骨骼
        if (!string.IsNullOrEmpty(_cfg.parentBonePath) && ctx.Actor != null)
            _parentTransform = ctx.Actor.transform.Find(_cfg.parentBonePath);

        CreateParticleInstance();
        UpdateTransform();

        if (_particleInstance != null)
        {
            _particleInstance.SetActive(true);
            foreach (var ps in _particleSystems)
            {
                if (ps != null)
                {
                    ps.Stop(true, ParticleSystemStopBehavior.StopEmittingAndClear);
                    ps.Play(false);
                }
            }
            _isPlaying = true;
            _lastTime = 0f;
        }

        CurrentPhase = Phase.Active;
    }

    public override void OnUpdate(FrameDriverContext ctx)
    {
        if (_cfg.updateTransform)
            UpdateTransform();

        if (!_isPlaying || _particleInstance == null) return;

        float currentTime = ctx.EventElapsedSeconds;
        if (_lastTime > currentTime)
        {
            // 时间倒流（不太可能在帧表里发生，但防御性处理）
            SimulateAll(currentTime, true);
        }
        else if (currentTime > _lastTime)
        {
            float dt = currentTime - _lastTime;
            SimulateAll(dt, false);
        }
        _lastTime = currentTime;
    }

    public override void OnStop(FrameDriverContext ctx, bool isFinish)
    {
        if (_particleInstance != null)
        {
            if (_cfg.destroyOnFinish)
            {
                if (Application.isPlaying)
                    UnityEngine.Object.Destroy(_particleInstance);
                else
                    UnityEngine.Object.DestroyImmediate(_particleInstance);
            }
            else
            {
                foreach (var ps in _particleSystems)
                    ps?.Stop(true, ParticleSystemStopBehavior.StopEmittingAndClear);
                _particleInstance.SetActive(false);
            }
        }
        _particleInstance = null;
        _particleSystems = null;
        _isPlaying = false;
        CurrentPhase = Phase.Finished;
    }

    private void CreateParticleInstance()
    {
        _particleInstance = UnityEngine.Object.Instantiate(_cfg.particlePrefab);
        _particleInstance.name = $"{_cfg.particlePrefab.name}_Effect";
        _particleInstance.hideFlags = HideFlags.HideInHierarchy;
        _particleSystems = new List<ParticleSystem>(
            _particleInstance.GetComponentsInChildren<ParticleSystem>());

        uint seed = _cfg.randomSeed == 0 ? (uint)UnityEngine.Random.Range(1, 10000) : _cfg.randomSeed;
        foreach (var ps in _particleSystems)
        {
            if (ps != null && ps.useAutoRandomSeed)
            {
                ps.useAutoRandomSeed = false;
                ps.randomSeed = seed++;
            }
        }
        _particleInstance.SetActive(false);
    }

    private void UpdateTransform()
    {
        if (_particleInstance == null) return;
        if (_parentTransform != null)
        {
            _particleInstance.transform.SetParent(_parentTransform, false);
            _particleInstance.transform.localPosition = _cfg.localPosition;
            _particleInstance.transform.localEulerAngles = _cfg.localRotation;
            _particleInstance.transform.localScale = _cfg.localScale;
        }
        else
        {
            _particleInstance.transform.position = _cfg.localPosition;
            _particleInstance.transform.eulerAngles = _cfg.localRotation;
            _particleInstance.transform.localScale = _cfg.localScale;
        }
    }

    private void SimulateAll(float time, bool restart)
    {
        if (_particleSystems == null) return;
        if (restart)
            foreach (var ps in _particleSystems)
                ps?.Simulate(0, true, true, false);

        float maxDt = Time.maximumDeltaTime;
        while (time > maxDt)
        {
            foreach (var ps in _particleSystems)
                ps?.Simulate(maxDt, true, false, false);
            time -= maxDt;
        }
        if (time > 0)
            foreach (var ps in _particleSystems)
                ps?.Simulate(time, true, false, false);
    }
}
```

### 1.3.14 ActionInstance.cs — 改造后新增 FrameEventRuntime 管理

```csharp
using System;
using UnityEngine;
using DeiveEx.TagTree;

public class ActionInstance
{
    public ActionAsset Config { get; }
    public ActionData RuntimeData { get; private set; }
    public ActionEventContext EventContext { get; private set; }
    public Actor Actor { get; private set; }

    /// <summary>本次播放的所有帧事件运行时实例。</summary>
    public FrameEventRuntime[] EventRuntimes { get; private set; }

    public ActionInstance(ActionAsset config)
    {
        Config = config;
        ResetRuntimeData();

        // 一次性创建全部事件的运行时实例——配置与状态从此刻分离
        var configs = config.FrameEvents;
        if (configs != null && configs.Count > 0)
        {
            EventRuntimes = new FrameEventRuntime[configs.Count];
            for (int i = 0; i < configs.Count; i++)
                EventRuntimes[i] = configs[i].CreateRuntime();
        }
        else
        {
            EventRuntimes = Array.Empty<FrameEventRuntime>();
        }
    }

    public void OnEnter(Actor actor, ActionEventContext context = default)
    {
        Actor = actor;
        EventContext = context;
        ApplyMotionConfig(context);

        var selfTags = Config.SelfTags;
        if (Actor != null && selfTags != null)
        {
            for (int i = 0; i < selfTags.Count; i++)
            {
                var tagRef = selfTags[i];
                if (tagRef == null) continue;
                Tag tagObj = tagRef.GetTag();
                if (tagObj != null)
                    Actor.AddTag(tagObj, ActorTagContainerType.Transient);
            }
        }
    }

    public void OnExit()
    {
        // 强制清理所有仍在活跃的事件（防泄漏）
        if (EventRuntimes != null && Actor != null)
        {
            var ctx = new FrameDriverContext { Actor = Actor, ActionInstance = this };
            for (int i = 0; i < EventRuntimes.Length; i++)
                EventRuntimes[i].ForceCleanup(ctx);
        }

        var selfTags = Config.SelfTags;
        if (Actor != null && selfTags != null)
        {
            for (int i = 0; i < selfTags.Count; i++)
            {
                var tagRef = selfTags[i];
                if (tagRef == null) continue;
                Tag tagObj = tagRef.GetTag();
                if (tagObj != null)
                    Actor.RemoveTag(tagObj, ActorTagContainerType.Transient);
            }
        }

        RestoreMotionConfig();
        Actor = null;
        EventContext = default;
    }

    // UpdateNormalizedTime / ResetRuntimeData / 运动策略 ... 保持不变
    // ...
}
```

### 1.3.15 ActionPlayer.cs — 改造后的帧驱动版本

```csharp
using UnityEngine;
using System;

/// <summary>
/// 帧驱动 ActionPlayer —— 替代 PlayableDirector，用整数帧号驱动所有 FrameEvent。
///
/// 核心优势：
///   - HitStop 只需不推进帧号，零成本
///   - 帧号 100% 整数精确，无浮点抖动
///   - 无 PlayableGraph 构建/销毁开销
/// </summary>
public class ActionPlayer : MonoBehaviour
{
    [SerializeField] private Actor _actor;

    /// <summary>期望播放速度。0=冻结（HitStop），1=正常。</summary>
    private float _expectedSpeed = 1f;

    /// <summary>亚帧累加器。每帧累加 deltaTime * speed * frameRate，溢出整数部分时推进帧号。</summary>
    private float _frameAccumulator;

    public float PlaybackSpeed => _expectedSpeed;
    public ActionInstance CurrentAction { get; private set; }
    public int CurrentFrame { get; private set; }
    public int CurrentFrameRate { get; private set; }
    public int TotalFrames { get; private set; }

    private ActionEventContext _currentContext;

    public event Action<ActionInstance> OnActionFinished;
    public event Action<ActionInstance> OnActionInterrupted;

    private void Awake()
    {
        if (_actor == null)
            _actor = GetComponentInParent<Actor>();
    }

    // ────────── 公共接口 ──────────

    public void StopAction()
    {
        var inst = CurrentAction;
        if (inst == null) return;

        CurrentAction = null;
        inst.OnExit();                   // 内部会 ForceCleanup 所有事件
        _actor?.ClearTransientTags();
        _currentContext = default;
        CurrentFrame = 0;
        CurrentFrameRate = 0;
        TotalFrames = 0;
        _frameAccumulator = 0f;
    }

    public void BeginAction(ActionAsset actionAsset, ActionEventContext context = default)
    {
        StopAction();
        _currentContext = context;

        if (actionAsset == null)
        {
            Debug.LogWarning("ActionPlayer: ActionAsset 为空", this);
            return;
        }

        CurrentAction = actionAsset.CreateActionInstance();
        CurrentAction.OnEnter(_actor, _currentContext);

        CurrentFrameRate = Mathf.Max(1, actionAsset.FrameRate);
        TotalFrames = actionAsset.TotalFrames;
        CurrentFrame = 0;
        _frameAccumulator = 0f;

        // 立即执行第 0 帧（与原来 Director.Evaluate() 对等）
        TickEvents(CurrentAction, 0);
    }

    public void SetSpeed(float speed)
    {
        _expectedSpeed = speed;
    }

    // ────────── 帧驱动核心 ──────────

    private void Update()
    {
        if (CurrentAction == null) return;

        // HitStop：速度为 0 时不推进帧号，所有事件自然冻结
        if (_expectedSpeed <= 0.0001f)
        {
            // 但仍然要让动画事件知道当前速度（用于冻结 Animancer State）
            BroadcastSpeedToAnimEvents();
            return;
        }

        // 亚帧累加：deltaTime * speed * frameRate → 累加器
        _frameAccumulator += Time.deltaTime * _expectedSpeed * CurrentFrameRate;

        // 每溢出一个整数帧就推进一次
        while (_frameAccumulator >= 1f)
        {
            _frameAccumulator -= 1f;
            CurrentFrame++;

            // 到达总帧数 → 结束或循环
            if (CurrentFrame >= TotalFrames)
            {
                if (CurrentAction.Config.IsLoop)
                {
                    CurrentAction.ResetRuntimeData();
                    // 停止所有当前事件，然后重新创建运行时实例
                    var ctx = MakeContext(CurrentFrame);
                    for (int i = 0; i < CurrentAction.EventRuntimes.Length; i++)
                        CurrentAction.EventRuntimes[i].ForceCleanup(ctx);

                    // 重新创建 EventRuntimes（因为循环需要干净的运行时状态）
                    var configs = CurrentAction.Config.FrameEvents;
                    var runtimes = new FrameEventRuntime[configs.Count];
                    for (int i = 0; i < configs.Count; i++)
                        runtimes[i] = configs[i].CreateRuntime();
                    // 用反射或增加一个内部方法来替换——这里建议在 ActionInstance 加一个方法
                    // CurrentAction.ReplaceEventRuntimes(runtimes); 
                    // 简化方案：直接重新 BeginAction
                    CurrentFrame = 0;
                    _frameAccumulator = 0f;
                    TickEvents(CurrentAction, 0);
                    return;
                }
                else
                {
                    // 正常结束
                    var finished = CurrentAction;
                    finished.OnExit();
                    _actor?.ClearTransientTags();
                    CurrentAction = null;
                    CurrentFrame = 0;
                    CurrentFrameRate = 0;
                    TotalFrames = 0;
                    _frameAccumulator = 0f;
                    OnActionFinished?.Invoke(finished);
                    return;
                }
            }

            TickEvents(CurrentAction, CurrentFrame);
        }

        // 非整数帧推进的帧里，仍需更新速度和持续事件
        UpdateActiveEvents(CurrentAction, CurrentFrame);

        // 更新归一化时间（供 ActionStateManager 的 CancelWindow 等使用）
        if (TotalFrames > 0)
            CurrentAction.UpdateNormalizedTime((double)CurrentFrame / TotalFrames);
    }

    // ────────── 事件驱动 ──────────

    /// <summary>推进到新帧时调用：检查所有事件的 Start/Stop/Update。</summary>
    private void TickEvents(ActionInstance inst, int frame)
    {
        var runtimes = inst.EventRuntimes;
        var configs = inst.Config.FrameEvents;

        for (int i = 0; i < runtimes.Length; i++)
        {
            var rt = runtimes[i];
            var cfg = configs[i];
            var ctx = MakeContext(frame, cfg);

            switch (rt.CurrentPhase)
            {
                case FrameEventRuntime.Phase.Idle:
                    if (frame >= cfg.startFrame)
                    {
                        rt.OnStart(ctx);
                        // 同帧到达 endFrame 的瞬时事件
                        if (frame >= cfg.endFrame)
                            rt.OnStop(ctx, true);
                        else
                            rt.OnUpdate(ctx);
                    }
                    break;

                case FrameEventRuntime.Phase.Active:
                    if (frame >= cfg.endFrame)
                        rt.OnStop(ctx, true);
                    else
                        rt.OnUpdate(ctx);
                    break;

                // Finished: 不做任何事
            }
        }
    }

    /// <summary>非帧推进的 Update 里更新所有活跃事件（主要是动画速度同步）。</summary>
    private void UpdateActiveEvents(ActionInstance inst, int frame)
    {
        var runtimes = inst.EventRuntimes;
        var configs = inst.Config.FrameEvents;

        for (int i = 0; i < runtimes.Length; i++)
        {
            if (runtimes[i].CurrentPhase == FrameEventRuntime.Phase.Active)
                runtimes[i].OnUpdate(MakeContext(frame, configs[i]));
        }
    }

    private void BroadcastSpeedToAnimEvents()
    {
        if (CurrentAction?.EventRuntimes == null) return;
        var runtimes = CurrentAction.EventRuntimes;
        var configs = CurrentAction.Config.FrameEvents;
        for (int i = 0; i < runtimes.Length; i++)
        {
            if (runtimes[i].CurrentPhase == FrameEventRuntime.Phase.Active)
                runtimes[i].OnUpdate(MakeContext(CurrentFrame, configs[i]));
        }
    }

    private FrameDriverContext MakeContext(int frame, FrameEventConfig cfg = null)
    {
        return new FrameDriverContext
        {
            Actor = _actor,
            ActionInstance = CurrentAction,
            CurrentFrame = frame,
            EventStartFrame = cfg?.startFrame ?? 0,
            EventEndFrame = cfg?.endFrame ?? 0,
            FrameRate = CurrentFrameRate,
            PlaybackSpeed = _expectedSpeed,
        };
    }
}
```

## 1.5 迁移数据的步骤

对于已有的 20 个 ActionAsset，需要把 Timeline Clip 的参数手工转移到 `_frameEvents` 列表：

1. 打开 Unity，选中一个 ActionAsset（如 `Sword_LightAttack_1`）
2. 打开对应的 Timeline 窗口，记录每个 Clip 的类型、起止时间、参数
3. 在 ActionAsset Inspector 的 `Frame Events` 列表里添加对应类型的 FrameEventConfig
4. `startFrame = round(clipStartTime * frameRate)`，`endFrame = round(clipEndTime * frameRate)`
5. 把 Clip 上的参数（TransitionAsset、ImpulseConfig、VelocityConfig 等）拖到对应的 Config 字段上
6. 对于 HitBox 的骨骼引用：打开 Timeline 看 ExposedReference 绑定的是哪个 Transform，记录其在 Hierarchy 里的路径，填到 `bonePath`
7. 填完后删除 ActionAsset 上的 `TimelineAsset` 引用（或者保留不用也行，不影响运行）

**建议顺序**：先迁移一个简单的（如 `Boxing_Idle`，只有一个 ContinuousAnimancerClip），验证帧驱动工作正常。再迁移一个攻击动作（如 `Sword_LightAttack_1`，有 Anim + HitBox + Velocity + Tag + Impulse），验证多事件协同。最后批量迁移其余。
